  event ProofRecorded(uint256 indexed campaignId, string proofHash);
  event WithdrawalRecorded(uint256 indexed campaignId, uint256 amount);

  function recordWithdrawal(...) external onlyAdmin { emit WithdrawalRecorded(...); }
