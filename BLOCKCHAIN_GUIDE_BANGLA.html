// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

contract TrustFundLedger {
  address public admin;

  event DonationRecorded(uint256 indexed campaignId, uint256 amount, address donor, string offchainTxId);
  event ProofRecorded(uint256 indexed campaignId, string proofHash);
  event WithdrawalRecorded(uint256 indexed campaignId, uint256 amount);

  modifier onlyAdmin() { require(msg.sender == admin, "Not authorized"); _; }
  constructor() { admin = msg.sender; }

  function recordDonation(...) external onlyAdmin { emit DonationRecorded(...); }
  function recordProof(...) external onlyAdmin { emit ProofRecorded(...); }
  function recordWithdrawal(...) external onlyAdmin { emit WithdrawalRecorded(...); }
}

RPC URL
আপনার অ্যাপ যে ওয়েব ঠিকানা দিয়ে ব্লকচেইনের সাথে "কথা বলে" — অনেকটা খাতাটির জন্য একটা API ঠিকানার মতো।
ABI
একটা ছোট ফাইল যেখানে লেখা থাকে স্মার্ট কন্ট্রাক্টের কোন কোন ফাংশন আছে, যাতে অন্য কোড সেটা কীভাবে ব্যবহার করবে তা বুঝতে পারে।
Transaction Hash (tx hash)
ব্লকচেইনে লেখা প্রতিটা এন্ট্রির একটা ইউনিক আইডি — অনেকটা রশিদ নম্বরের মতো, যে কেউ খুঁজে বের করতে পারে।
Mainnet
আসল, লাইভ নেটওয়ার্ক, যেখানে লেনদেনে আসল টাকা ব্যবহার হয়।
Testnet / Sandbox
প্র্যাকটিসের জন্য নেটওয়ার্ক, যেখানে নকল/ফ্রি টেস্ট মুদ্রা ব্যবহার হয়। Sepolia হলো Ethereum-এর Testnet।
Faucet
একটা ফ্রি ওয়েবসাইট যা কোনো ওয়ালেট ঠিকানায় সামান্য টেস্ট মুদ্রা পাঠিয়ে দেয়, Testnet-এ প্র্যাকটিস করার জন্য।
MetaMask
সবচেয়ে জনপ্রিয় ওয়ালেট অ্যাপ — একটা ব্রাউজার এক্সটেনশন যা আপনার চাবি সংরক্ষণ করে ও লেনদেন অনুমোদনে সাহায্য করে।
Gas Fee
ব্লকচেইনে কোনো নতুন এন্ট্রি লিখতে যে ছোট ফি দিতে হয়।
Solidity
